\newcommand{\screenshot}[3]{
\begin{figure}[ht]
 \centering
\hspace*{-.05\textwidth}
\makebox[\textwidth]{
  \includegraphics[width = 19cm]{#3}}
 \caption{#1} \label{fig:#2} 
\end{figure}
\FloatBarrier
}


\newcommand{\screenshotWidth}[4]{

  \includegraphics[width = {#4 cm}]{#3}}
 \caption{#1.} \label{fig:#2} 
\end{figure}
\FloatBarrier
}
